dict={0:[2,3],2:[10],3:[4,5],5:[6],8:[9]}

``` python
# dic : 字典 {key:value}
# if a in dic:


dic={0:[2,3],2:[10],3:[4,5],5:[6],8:[9]}

queue = [0] #看誰的周圍

while queue:    #當所有的點看完時，queue就空了
    node = queue.pop(0)     #刪除索引值0，並傳出
    print(node)

    if node in dic:
        for i in dic[node]:
            queue.append(i)
```
